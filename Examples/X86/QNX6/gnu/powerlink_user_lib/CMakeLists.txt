# Configure depending selected mode
#

        SET(OBJDICT CiA302-4_MN)

                        ${POWERLINK_SOURCE_DIR}/EplStack/EplObdCdc.c
                        ${POWERLINK_SOURCE_DIR}/EplStack/EplCfmu.c)

    SET(OBJDICT Qt_MN)
